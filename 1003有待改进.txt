int* primenum(int* a,int b){
	int i,j=0,n,flag;
	a[j]=2;
	for(i=3;i<=b;i++){
		flag=1;
		for(n=0;(a[n]*a[n])<=i;n++)
			if(i%a[n]==0) flag=0;
			if(flag==1){
				j++;
				a[j]=i;
			}
	}
	j++;
	a[j]=0;
	return(a);
}
