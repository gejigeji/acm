#include  <stdio.h>
int hash[10]={0};

int main () {
	FILE *fin = fopen ("crypt1.in", "r"); 
	FILE *fout = fopen ("crypt1.out", "w");

	int i,j,m,n,p,N,num[9]={0};
	int s1,s2,s3,s4,s5,temp;
	int ans=0,t1,t2,t3,pos,flag;

	fscanf(fin,"%d",&N);
	for(i=0;i<N;i++){
		fscanf(fin,"%d",&num[i]);
		hash[num[i]]=1;
	}
	for(j=0;j<N-1;j++){
		flag=1;
		for(i=0;i<N-1;i++){
			if(num[i+1]<num[i]){
				temp=num[i];
				num[i]=num[i+1];
				num[i+1]=temp;
				flag=0;
			}
		}
		if(flag==1) break;
	}

	for(i=0;i<N;i++){
		s1=num[i];
		for(j=0;j<N;j++){
			s2=num[j];
			if(s1*s2>=10) break;
			for(m=0;m<N;m++){
				s3=num[m];
				if(s1*s3>=10) break;
				for(n=0;n<N;n++){
					s4=num[n];
					if((s1*s2*10+s4*s2>=100)||(s1*s3*10+s4*s3>=100)) break;
					for(p=0;p<N;p++){
						s5=num[p];
						t1=s3*s1*100+s3*s4*10+s3*s5;
						t2=s2*s1*100+s2*s4*10+s2*s5;
						t3=t1+t2*10;
						if(t1>=1000||t2>=1000||t3>=10000) break;

						flag=1;
						while(t1>0){
							pos=t1%10;
							if(!hash[pos]) {flag=0;break;}
							t1=t1/10;
						}
						if(flag==0) continue;
						while(t2>0){
							pos=t2%10;
							if(!hash[pos]) {flag=0;break;}
							t2=t2/10;
						}
						if(flag==0) continue;
						while(t3>0){
							pos=t3%10;
							if(!hash[pos]) {flag=0;break;}
							t3=t3/10;
						}
						if(flag==0) continue;
						ans++;
					}
				}
			}
		}
	}

	fprintf(fout,"%d\n",ans);
	fclose(fin);
	fclose(fout);
	exit (0); 
}
