#include <stdio.h>
#include <strings.h>

void main(){
	int i=0,nn,mm;
	char a[100];
	while(1){
		scanf("%s",a);
		if(a[0]=='') break;
		if(a[0]=='(')
			printf("NO\n");
		while(a[i]!='\n'){
			if(a[i]=='(') nn++;
			else mm++;
			i++;
		}
		if(nn!=mm)
			printf("NO\n");	
		else 
			printf("YES\n");
	}
	exit (0);
}
